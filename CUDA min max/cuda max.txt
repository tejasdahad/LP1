using namespace std;



__global__ void maxi(int *a,int *b,int n)
{


	int block=256*blockIdx.x;


	int max=0;

	for(int i=block;i<min(256+block,n);i++)
	{


		if(max<a[i])
		{

			max=a[i];
	
		}





	}
	b[blockIdx.x]=max;

}

{

int n =10000;
  int a[n];

	cudaEvent_t start,end;

	cudaMalloc(&ad,size);

	int grids=ceil(n*1.0f/256.0f);
	cudaMalloc(&bd,grids*sizeof(int));
